26 inline Eigen::Isometry3d vectorToIsometry(std::vector<double> &poseVector) {
27  double *ptr = &poseVector[0];
28  Eigen::Map<Eigen::VectorXd> p(ptr, 7);
29 
30  Eigen::Isometry3d pose(Eigen::Isometry3d::Identity());
31  pose.translation() = p.head(3);
32  // w, x, y, z
33  Eigen::Quaterniond q(p[6], p[3], p[4], p[5]);
34  pose.linear() = Eigen::Matrix3d(q);
35  return pose;
36 }
